// ── CORE FUNCTIONS ────────────────────────────────────────
function getMonthKey(date) {
  return `${date.getFullYear()}-${String(date.getMonth()+1).padStart(2,'0')}`;
}

function getMonthExpenses(data, date) {
  const key = getMonthKey(date);
  return data.filter(e => e.date.startsWith(key));
}

function changeMonth(dir) {
  currentDate = new Date(currentDate.getFullYear(), currentDate.getMonth() + dir, 1);
  activeFilter = "All";
  render();
}

function saveBudget() {
  const b = parseFloat(document.getElementById('fBudget').value);
  if (!isNaN(b) && b > 0) {
    budget = b;
    localStorage.setItem('pp_budget', b);
    render();
  }
}

function addExpense() {
  const desc = document.getElementById('fDesc').value.trim();
  const amt  = parseFloat(document.getElementById('fAmt').value);
  const date = document.getElementById('fDate').value;
  const cat  = document.getElementById('fCat').value;
  const type = document.getElementById('fType').value;

  if (!desc || isNaN(amt) || amt <= 0 || !date) {
    alert('Please fill in description, amount, and date.');
    return;
  }

  const data = loadData();
  const id = Date.now();
  data.push({ id, desc, amt, date, cat, type });
  saveData(data);

  document.getElementById('fDesc').value = '';
  document.getElementById('fAmt').value = '';
  render();
}

function deleteExpense(id) {
  const data = loadData().filter(e => e.id !== id);
  saveData(data);
  render();
}

function setFilter(f) {
  activeFilter = f;
  render();
}

// ── RENDER ────────────────────────────────────────────────
function render() {
  const data     = loadData();
  const monthExp = getMonthExpenses(data, currentDate);
  const months   = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];

  // Budget
  const savedBudget = localStorage.getItem('pp_budget');
  if (savedBudget) {
    budget = parseFloat(savedBudget);
    document.getElementById('fBudget').value = budget;
  }

  // Month label
  document.getElementById('monthLabel').textContent =
    `${months[currentDate.getMonth()]} ${currentDate.getFullYear()}`;

  // Totals
  const total     = monthExp.reduce((s,e) => s + e.amt, 0);
  const essential = monthExp.filter(e => e.type === 'Essential').reduce((s,e) => s + e.amt, 0);
  const lifestyle = monthExp.filter(e => e.type === 'Lifestyle').reduce((s,e) => s + e.amt, 0);
  const left      = budget - total;

  document.getElementById('sumTotal').textContent     = fmt(total);
  document.getElementById('sumEssential').textContent = fmt(essential);
  document.getElementById('sumLifestyle').textContent = fmt(lifestyle);
  document.getElementById('savingsAmt').textContent   = fmt(Math.max(0, left));
  document.getElementById('savingsPct').textContent   =
    budget > 0 ? `${Math.round((left/budget)*100)}% of ₹${fmt(budget)} budget` : 'Set budget above';

  // Category breakdown
  const catMap = {};
  monthExp.forEach(e => {
    catMap[e.cat] = (catMap[e.cat] || 0) + e.amt;
  });
  const maxCat = Math.max(...Object.values(catMap), 1);
  const catSorted = Object.entries(catMap).sort((a,b) => b[1]-a[1]);

  document.getElementById('catList').innerHTML = catSorted.length ? catSorted.map(([cat, amt]) => `
    <div class="cat-item" onclick="setFilter('${cat}')">
      <div class="cat-dot" style="background:${CAT_COLORS[cat]||'#999'}"></div>
      <div class="cat-name">${cat}</div>
      <div class="cat-bar-wrap"><div class="cat-bar" style="width:${Math.round((amt/maxCat)*100)}%;background:${CAT_COLORS[cat]||'#999'}"></div></div>
      <div class="cat-amt">${fmt(amt)}</div>
    </div>`).join('')
  : '<div class="empty">No expenses this month</div>';

  // Bar chart — last 6 months
  const barData = [];
  for (let i = 5; i >= 0; i--) {
    const d = new Date(currentDate.getFullYear(), currentDate.getMonth() - i, 1);
    const exp = getMonthExpenses(data, d);
    const tot = exp.reduce((s,e) => s + e.amt, 0);
    barData.push({ label: months[d.getMonth()], amt: tot });
  }
  const maxBar = Math.max(...barData.map(b => b.amt), 1);
  document.getElementById('barChart').innerHTML = barData.map((b, i) => {
    const h = Math.round((b.amt / maxBar) * 110);
    const isActive = i === 5;
    return `<div class="bar-col">
      <div class="bar-amt">${b.amt > 0 ? fmtK(b.amt) : ''}</div>
      <div class="bar-block" style="height:${h}px;background:${isActive?'#1F3864':'#9ab0cc'}" data-tip="${b.label}: ${fmt(b.amt)}"></div>
      <div class="bar-month">${b.label}</div>
    </div>`;
  }).join('');

  // Filter bar
  const cats = ['All', ...new Set(monthExp.map(e => e.cat))];
  document.getElementById('filterBar').innerHTML = cats.map(c =>
    `<button class="filter-btn ${activeFilter===c?'active':''}" onclick="setFilter('${c}')">${c}</button>`
  ).join('');

  // Transactions
  const filtered = activeFilter === 'All' ? monthExp
    : monthExp.filter(e => e.cat === activeFilter);
  const sorted = [...filtered].sort((a,b) => new Date(b.date) - new Date(a.date));

  document.getElementById('txnBody').innerHTML = sorted.length
    ? sorted.map(e => `
      <tr>
        <td>${formatDate(e.date)}</td>
        <td>${e.desc}</td>
        <td><span class="txn-cat"><span class="txn-dot" style="background:${CAT_COLORS[e.cat]||'#999'}"></span>${e.cat}</span></td>
        <td><span style="font-size:11px;padding:2px 8px;border-radius:10px;background:${e.type==='Essential'?'#ffeaea':e.type==='Lifestyle'?'#fff8e1':'#eef5ff'};color:${e.type==='Essential'?'#c0392b':e.type==='Lifestyle'?'#e67e22':'#1F3864'}">${e.type}</span></td>
        <td class="txn-amt" style="text-align:right">${fmt(e.amt)}</td>
        <td><button class="delete-btn" onclick="deleteExpense(${e.id})">✕</button></td>
      </tr>`)
    .join('')
    : `<tr><td colspan="6" class="empty">No transactions. Add your first expense!</td></tr>`;

  // Insights
  renderInsights(monthExp, total, essential, lifestyle);
}

function renderInsights(expenses, total, essential, lifestyle) {
  const insights = [];
  const cats = {};
  expenses.forEach(e => cats[e.cat] = (cats[e.cat] || 0) + e.amt);

  // Over budget
  if (budget > 0 && total > budget) {
    insights.push({ type: 'warn', title: `Over budget by ${fmt(total - budget)}`,
      msg: `You've spent ${fmt(total)} against a budget of ${fmt(budget)}. Consider reviewing discretionary spending.` });
  } else if (budget > 0 && total > budget * 0.85) {
    insights.push({ type: 'warn', title: 'Approaching budget limit',
      msg: `You've used ${Math.round((total/budget)*100)}% of your monthly budget. ${fmt(budget-total)} remaining.` });
  }

  // Ecom spike
  const ecom = cats['Ecommerce / Shopping'] || 0;
  if (ecom > 5000) {
    insights.push({ type: 'tip', title: `Ecommerce at ${fmt(ecom)} — Optimise Timing`,
      msg: `Large ecom spends this month. Best time to buy on Amazon/Flipkart: end of month sales (last 3 days) or major sale events. Use HDFC/ICICI card for extra 5-10% off on selected days.` });
  }

  // Utility tip
  const elec = cats['Electricity'] || 0;
  if (elec > 2500) {
    insights.push({ type: 'tip', title: `Electricity at ${fmt(elec)} — Reduce by 15-20%`,
      msg: `Shift high-consumption appliances (AC, geyser, washing machine) to off-peak hours (11pm–6am). Most DISCOMs charge lower ToD tariffs in off-peak slots.` });
  }

  // Food tip
  const food = cats['Food / Dining'] || 0;
  if (food > 2000) {
    insights.push({ type: 'tip', title: `Food delivery at ${fmt(food)} — Save up to 40%`,
      msg: `Swiggy & Zomato offer best discounts between 2pm-5pm and after 10pm. Use Swiggy One or Zomato Gold for free delivery. Weekend cooking saves ₹800-1200/month.` });
  }

  // EMI check
  const emi = cats['EMI / Loan'] || 0;
  if (emi > 0) {
    const emiPct = Math.round((emi/total)*100);
    if (emiPct > 40) {
      insights.push({ type: 'warn', title: `EMI at ${emiPct}% of total spend — High FOIR`,
        msg: `Your Fixed Obligation to Income Ratio appears high. Ideal FOIR for loan eligibility is below 40%. Consider prepaying one loan partially to reduce monthly obligation.` });
    } else {
      insights.push({ type: 'good', title: `EMI-to-spend ratio looks healthy at ${emiPct}%`,
        msg: `Your loan EMI is within manageable range. Keep maintaining on-time payments to build your credit score above 750.` });
    }
  }

  // OTT
  const ott = cats['OTT / Subscriptions'] || 0;
  if (ott > 1000) {
    insights.push({ type: 'tip', title: `Subscriptions at ${fmt(ott)} — Audit annually`,
      msg: `Check for unused OTT subscriptions. Most platforms offer annual plans at 30-40% discount vs monthly. Share family plans across 4 accounts for Netflix, Hotstar, Amazon.` });
  }

  // Good month
  if (budget > 0 && total < budget * 0.75) {
    insights.push({ type: 'good', title: `Good month — ${Math.round(((budget-total)/budget)*100)}% under budget!`,
      msg: `You saved ${fmt(budget-total)} this month. Consider moving this to a liquid fund or SIP rather than letting it sit in a savings account earning 3-4%.` });
  }

  // Fuel
  const fuel = cats['Car / Fuel'] || 0;
  if (fuel > 2500) {
    insights.push({ type: 'tip', title: `Fuel at ${fmt(fuel)} — Fill on Tuesdays & Wednesdays`,
      msg: `Petrol prices in India update every morning. Fill up early morning (before 6am) or on weekdays when prices stabilise. HPCL, BPCL credit card users get ₹100-150/fill savings.` });
  }

  document.getElementById('insightsArea').innerHTML = insights.length
    ? insights.map(i => `<div class="insight-card ${i.type}"><h4>${i.title}</h4><p>${i.msg}</p></div>`).join('')
    : `<div class="insight-card good"><h4>✅ Looking good this month!</h4><p>Add more expenses to get personalised optimisation insights.</p></div>`;
}

// ── UTILS ─────────────────────────────────────────────────
function fmt(n) {
  return '₹' + Number(n).toLocaleString('en-IN');
}
function fmtK(n) {
  return n >= 100000 ? `₹${(n/100000).toFixed(1)}L` : n >= 1000 ? `₹${(n/1000).toFixed(1)}K` : `₹${n}`;
}
function formatDate(d) {
  const dt = new Date(d);
  return `${dt.getDate()} ${['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'][dt.getMonth()]}`;
}
